# covert go struct 2 typescript interface inspired by [go2ts](https://github.com/skia-dev/go2ts/)

## Install

    go get github.com/xx1906/go2ts

## Example

Input:

```go
import (
	"image/color"
	"log"
	"os"

	"github.com/xx1906/go2ts"
)

func main() {
	type Direction string

	const (
		Up    Direction = "up"
		Down  Direction = "down"
		Left  Direction = "left"
		Right Direction = "right"
	)

	AllDirections := []Direction{Up, Down, Left, Right}

	type Position struct {
		X int
		Y int
	}

	type Turtle struct {
		Position  `json:"Coordinates"`
		Color     color.Alpha
		Direction Direction
	}

	generator := go2ts.New()
	generator.Add(Turtle{})
	generator.AddUnion(AllDirections)
	generator.Render(os.Stdout)
}
```

Output:

```ts
// DO NOT EDIT. This file is automatically generated.

export interface Position {
  X: number;
  Y: number;
}

export interface Alpha {
  A: number;
}

export interface Turtle {
  Coordinates: Position;
  Color: Alpha;
  Direction: Direction;
}

export type Direction = 'up' | 'down' | 'left' | 'right';
```
